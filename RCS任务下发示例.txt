public class RequestCreateTaskByRCS
{
    /// <summary>
    /// 系统token
    /// </summary>
    public string SysToken { get; set; }

    /// <summary>
    /// 单号
    /// </summary>
    public string ReceiveTaskID { get; set; }

    /// <summary>
    /// 地图编码
    /// </summary>
    public string MapCode { get; set; }

    /// <summary>
    /// 任务模板编码
    /// </summary>
    public string TaskCode { get; set; }

    /// <summary>
    /// 车辆集群分组编码，标识调用指定集群内车辆执行任务，可设置为空
    /// </summary>
    public string AgvGroupCode { get; set; }

    /// <summary>
    /// 车辆编码，标识调用指定车辆执行任务，可设置为空
    /// </summary>
    public string AGVCode { get; set; }

    /// <summary>
    /// 任务优先级 1~10，默认为5
    /// </summary>
    public int Priority { get; set; }

    /// <summary>
    /// 任务参数数组，json格式字符串，格式参考示例
    /// </summary>
    [SugarColumn(IsJson = true)]
    public List<Variables> Variables { get; set; }
}
/// <summary>
/// RCS返回
/// </summary>
[TableConfiguration(false, true)]
public class SerializeReply
{
    public bool Success { get; set; }

    public string Content { get; set; }
}

Public class Main
{
	public void AddTask()
	{
		 var url = $"http://127.0.0.1:50060/Task/CreateTask";
		 var requestTask = new RequestCreateTaskByRCS();
		 requestTask.SysToken="外接系统标识";
		 requestTask.ReceiveTaskID="单号";
		 requestTask.MapCode="地图编码";
		 requestTask.TaskCode="任务模版编号";
		 requestTask.Variables.Add(new Variables(){
			Code="StartPoint",
			Value="PNT_1_11",
		 });
		 result = HttpClientHelper.ExecuteAsync<SerializeReply>(url, HttpMethod.Post, JsonConvert.SerializeObject(requestTask), null!).Result;
		 CommonFunction.WriteLog($"入库任务下发反馈结果：\r\n{JsonConvert.SerializeObject(result)}", "入库任务下发");
	}
}
